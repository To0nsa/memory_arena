97void arena_stats_record_growth(t_arena_stats* stats, size_t new_size)
98{
99 if (!stats)
100 return;
101
102 size_t count = stats->growth_history_count;
103 size_t* new_buffer = realloc(stats->growth_history, (count + 1) * sizeof(size_t));
104 if (!new_buffer)
105 {
106 ALOG("arena_stats_record_growth: failed to realloc for growth history\n");
107 return;
108 }
109
110 stats->growth_history = new_buffer;
111 stats->growth_history[count] = new_size;
112 stats->growth_history_count++;
113}
