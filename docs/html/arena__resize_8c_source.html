59bool arena_grow(t_arena* arena, size_t required_size)
60{
61 if (!arena)
62 return false;
63
64 if (required_size == 0)
65 return true;
66
67 ARENA_LOCK(arena);
68
69 if (!arena_grow_validate(arena, required_size))
70 {
71 ARENA_UNLOCK(arena);
72 return false;
73 }
74
75 size_t old_size = arena->size;
76 size_t new_size = arena_grow_compute_new_size(arena, required_size);
77 if (new_size == 0)
78 {
79 arena_report_error(arena, "arena_grow failed: computed size invalid");
80 ARENA_UNLOCK(arena);
81 return false;
82 }
83
84 if (new_size > (size_t)(1ULL << 32)) // 4GiB
85 {
86 arena_report_error(arena, "arena_grow rejected size: %zu", new_size);
87 return false;
88 }
89 bool success = arena_grow_realloc_buffer(arena, new_size, old_size);
90 ARENA_UNLOCK(arena);
91 return success;
92}

108bool arena_might_shrink(t_arena* arena)
109{
110 if (!should_attempt_shrink(arena))
111 return false;
112
113 ARENA_LOCK(arena);
114 ARENA_CHECK(arena);
115

121 const size_t target = arena_shrink_target(used);

123 if (target < size)

276static inline size_t arena_grow_compute_new_size(t_arena* arena, size_t required_size)
277{
278 arena_grow_callback cb = arena->grow_cb ? arena->grow_cb : default_grow_cb;
279 size_t requested = arena->offset + required_size;
280 size_t new_size = cb(arena->size, required_size);

363static inline bool arena_can_shrink(t_arena* arena, size_t new_size)
364{
365 if (!arena)
366 return false;
367
368 bool owns = atomic_load_explicit(&arena->owns_buffer, memory_order_acquire);
369 bool grow = atomic_load_explicit(&arena->can_grow, memory_order_acquire);
370
371 if (!owns || !grow)
372 return false;
373
374 if (new_size < arena->offset)
375 return false;
376
377 if (new_size == arena->offset)
378 return true;
379
380 double shrink_ratio = (double) new_size / (double) arena->size;
381 return shrink_ratio <= ARENA_MIN_SHRINK_RATIO;
382}
