142bool arena_grow(t_arena* arena, size_t required_size)
143{
144 if (!arena)
145 return false;
146
147 if (required_size == 0)
148 return true;
149
150 ARENA_LOCK(arena);
151
152 if (!arena_grow_validate(arena, required_size))
153 {
154 ARENA_UNLOCK(arena);
155 return false;
156 }
157
158 size_t old_size = arena->size;
159 size_t new_size = arena_grow_compute_new_size(arena, required_size);
160 if (new_size == 0)
161 {
162 arena_report_error(arena, "arena_grow failed: computed size invalid");
163 ARENA_UNLOCK(arena);
164 return false;
165 }
166
167 if (new_size > (size_t)(1ULL << 32)) // 4GiB
168 {
169 arena_report_error(arena, "arena_grow rejected size: %zu", new_size);
170 return false;
171 }
172 bool success = arena_grow_realloc_buffer(arena, new_size, old_size);
173 ARENA_UNLOCK(arena);
174 return success;
175}

384bool arena_might_shrink(t_arena* arena)
385{

388

553{
554 arena_grow_callback cb = arena->grow_cb ? arena->grow_cb : default_grow_cb;

594{

596 if (!new_buf)

639static inline bool arena_can_shrink(t_arena* arena, size_t new_size)
640{
641 if (!arena)
642 return false;
643
644 bool owns = atomic_load_explicit(&arena->owns_buffer, memory_order_acquire);
645 bool grow = atomic_load_explicit(&arena->can_grow, memory_order_acquire);
646
647 if (!owns || !grow)
648 return false;
649
650 if (new_size < arena->offset)
651 return false;
652
653 if (new_size == arena->offset)
654 return true;
655
656 double shrink_ratio = (double) new_size / (double) arena->size;
657 return shrink_ratio <= ARENA_MIN_SHRINK_RATIO;
658}
