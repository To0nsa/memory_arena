Using thread-local scratch arenas for parallel tasks.

Using thread-local scratch arenas for parallel tasks.This example demonstrates how to use get_thread_scratch_arena() in a multithreaded environment to perform fast, thread-isolated memory allocations. Each thread gets its own t_arena, reset automatically between uses. No synchronization is required between threads.

This pattern is ideal for:

  • Multithreaded render pipelines
  • Parsers and serializers
  • Job systems that need temporary buffers
#include "arena.h"
#include <pthread.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#define NUM_THREADS 4
void* thread_func(void* arg)
{
int thread_id = *(int*)arg;
t_arena* arena = get_thread_scratch_arena();
if (!arena)
{
fprintf(stderr, "Thread %d: failed to get scratch arena\n", thread_id);
return NULL;
}
// Allocate a temporary string buffer
char* buffer = (char*) arena_alloc(arena, 256);
snprintf(buffer, 256, "Hello from thread %d!", thread_id);
// Simulate work
printf("%s\n", buffer);
usleep(100000); // simulate delay
// Memory is automatically reset after each call to get_thread_scratch_arena()
return NULL;
}
int main(void)
{
pthread_t threads[NUM_THREADS];
int ids[NUM_THREADS];
for (int i = 0; i < NUM_THREADS; ++i)
{
ids[i] = i;
pthread_create(&threads[i], NULL, thread_func, &ids[i]);
}
for (int i = 0; i < NUM_THREADS; ++i)
pthread_join(threads[i], NULL);
return 0;
}

{
if (!thread_scratch_initialized)
{
