198int wait_for_key(t_arena_visualizer* vis)
199{
200 while (1)
201 {
202 draw_visualizer(vis);
203 uint32_t key = notcurses_get_blocking(vis->nc, NULL);
204 switch (key)
205 {
206 case 'q':
207 case 'Q':
208 case NCKEY_ESC:
209 return 0; // user wants to quit
210 case NCKEY_UP:
211 vis->scroll_offset++;
212 break;
213 case NCKEY_DOWN:
214 vis->scroll_offset--;
215 break;
216 case 'n':
217 case 'N':
218 return 1; // next step
219 default:
220 // do nothing
221 break;
222 }
223 }
224}

232static void interactive_loop(t_arena_visualizer* vis)
233{
234 // We won't do an internal scenario script here — we assume user code calls

257void arena_start_interactive_visualizer(t_arena_visualizer* vis, t_arena* arena)
258{
259 if (!vis || !arena)
260 return;
261
262 struct notcurses_options opts = {
263 .termtype = NULL,
264 .loglevel = NCLOGLEVEL_SILENT,
265 };
266
267 vis->nc = notcurses_init(&opts, NULL);
268 if (!vis->nc)
269 {
270 fprintf(stderr, "[Visualizer] Failed to init notcurses\n");
271 return;
272 }
273
274 vis->main_plane = notcurses_stdplane(vis->nc);
275 pthread_mutex_init(&vis->lock, NULL);
276
277 atomic_store(&vis->history_index, 0);
278 atomic_store(&vis->history_count, 0);
279
280 vis->scroll_offset = 0;
281 vis->script_done = false;
282 vis->arena = arena;
283
284 // We won't do an infinite loop here. We'll just do a single call
285 // to interactive_loop so user can press 'n' or 'q'
286 // Then we return control to main. So main can do partial steps, come back, etc.
287
288 interactive_loop(vis);
289 // user pressed q or n => we return
290}
