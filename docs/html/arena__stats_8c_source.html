164void arena_stats_record_growth(t_arena_stats* stats, size_t new_size)
165{
166 if (!stats)
167 return;
168
169 size_t count = stats->growth_history_count;
170 size_t* new_buffer = realloc(stats->growth_history, (count + 1) * sizeof(size_t));
171 if (!new_buffer)
172 {
173 ALOG("arena_stats_record_growth: failed to realloc for growth history\n");
174 return;
175 }
176
177 stats->growth_history = new_buffer;
178 stats->growth_history[count] = new_size;
179 stats->growth_history_count++;
180}
