283void arena_pop(t_arena* arena, t_arena_marker marker)
284{
285 if (!arena)
286 return;
287
288 ARENA_LOCK(arena);
289 if (marker > arena->offset)

291 arena_report_error(arena, "arena_pop failed: invalid marker %zu (offset: %zu)", marker, arena->offset);

293 return;
294 }
295
296 arena_poison_memory(arena->buffer + marker, arena->offset - marker);
