116static inline bool arena_alloc_validate_input(t_arena* arena, size_t size, size_t alignment, const char* label)
117{
118 if (!arena)
119 {
120 arena_report_error(NULL, "%s failed: NULL arena", label);
121 return false;
122 }
123 if (size == 0)
124 {
125 arena_report_error(arena, "%s failed: zero-size allocation", label);
126 return false;
127 }
128 if (alignment == 0 || (alignment & (alignment - 1)) != 0)
129 {
130 arena_report_error(arena, "%s failed: alignment (%zu) is not a power-of-two", label, alignment);
131 return false;
132 }
133 return true;
134}

308static inline bool arena_ensure_capacity(t_arena* arena, size_t size, size_t alignment, const char* label,
309 size_t* aligned_offset, size_t* wasted)
310{
311 *aligned_offset = arena_calc_aligned_offset(arena, alignment);
312 *wasted = *aligned_offset - arena->offset;
313
314 if (*aligned_offset + size <= arena->size)
315 return true;
316
317 if (!arena_try_grow(arena, size, label))
318 return false;
319
320 *aligned_offset = arena_calc_aligned_offset(arena, alignment);
321 *wasted = *aligned_offset - arena->offset;
322
323 return *aligned_offset + size <= arena->size;
324}

356static inline void arena_update_stats(t_arena* arena, size_t size, size_t wasted)
357{
358 if (!arena)
359 return;
360
361 ARENA_LOCK(arena);
362 arena->stats.allocations++;
363 arena->stats.live_allocations++;
364 arena->stats.bytes_allocated += size;
365 arena->stats.wasted_alignment_bytes += wasted;
366 arena->stats.alloc_id_counter++;
367 arena->stats.last_alloc_size = size;
368 arena->stats.last_alloc_offset = arena->offset - size;

526void* arena_alloc_internal(t_arena* arena, size_t size, size_t alignment, const char* label)
527{
528 if (!arena_alloc_validate_input(arena, size, alignment, label))
529 return NULL;
530
531 ARENA_LOCK(arena);
532 ARENA_CHECK(arena);
533
534 if (arena_is_being_destroyed(arena, label))
535 {
536 ARENA_UNLOCK(arena);
537 return NULL;
538 }
539
540 if (arena_check_overflow(arena, size, label))
541 {
542 ARENA_UNLOCK(arena);
543 return NULL;
544 }
545
546 size_t aligned_offset = 0;
547 size_t wasted = 0;
548 if (!arena_ensure_capacity(arena, size, alignment, label, &aligned_offset, &wasted))
549 {
550 arena->stats.failed_allocations++;
551 arena_report_error(arena, "%s failed: out of memory (requested: %zu)", label, size);
552 ARENA_UNLOCK(arena);
553 return NULL;
554 }
555
556 arena_commit_allocation(arena, size, wasted, aligned_offset);
557 void* result = arena->buffer + aligned_offset;
558 arena_zero_if_needed(result, size, label);
559 arena_invoke_allocation_hook(arena, result, size, aligned_offset, wasted, label);
560
561 ALOG("[arena] %s: Allocated %zu bytes @ offset %zu (arena %p)\n", label, size, aligned_offset, (void*) arena);
562
563 ARENA_CHECK(arena);
564 ARENA_UNLOCK(arena);
565 return result;
566}
