52void* arena_realloc_last(t_arena* arena, void* old_ptr, size_t old_size, size_t new_size)
53{
54 if (!arena_realloc_validate(arena, old_ptr, new_size))
55 return NULL;
56
57 ARENA_LOCK(arena);
58 ARENA_CHECK(arena);
59
60 bool is_last = is_last_allocation(arena, old_ptr, old_size);
61 void* result = is_last ? realloc_in_place(arena, old_ptr, old_size, new_size) : NULL;
62
63 ARENA_UNLOCK(arena);
64
65 return is_last ? result : realloc_fallback(arena, old_ptr, old_size, new_size);
66}

214static inline void* realloc_in_place(t_arena* arena, void* old_ptr, size_t old_size, size_t new_size)
215{
216 size_t new_end = arena->offset - old_size + new_size;
217
218 if (new_end > arena->size && !arena_grow(arena, new_size - old_size))
219 return arena_report_error(arena, "arena_realloc_last failed: growth failed (needed %zu bytes)",
220 new_size - old_size),
221 NULL;
222
223 if (new_size < old_size)
224 arena_poison_memory((uint8_t*) old_ptr + new_size, old_size - new_size);
225
226 update_realloc_stats(arena, old_ptr, new_size, old_size, "arena_realloc_last (in-place)");
227 return old_ptr;
