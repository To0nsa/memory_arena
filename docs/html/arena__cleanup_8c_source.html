186static inline void arena_free_buffer_if_owned(t_arena* arena)
187{
188 bool owns = atomic_load_explicit(&arena->owns_buffer, memory_order_acquire);
189 if (owns && arena->buffer)
190 {
191 arena_poison_memory(arena->buffer, arena->size);
192 free(arena->buffer);
193 arena->buffer = NULL;
194 atomic_store_explicit(&arena->owns_buffer, false, memory_order_release);
195 }
196}
