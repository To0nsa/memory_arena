213bool scratch_pool_init(t_scratch_arena_pool* pool, size_t slot_size, bool thread_safe)
214{
215 if (!pool || slot_size == 0)
216 return arena_report_error(NULL, "scratch_pool_init failed: invalid arguments"), false;
217
218 memset(pool, 0, sizeof(*pool));
219 pool->slot_size = slot_size;
220 pool->thread_safe = thread_safe;
221
222#ifdef ARENA_ENABLE_THREAD_SAFE
223 if (thread_safe)
224 pthread_mutex_init(&pool->lock, NULL);
225#endif
226
227 for (int i = 0; i < SCRATCH_MAX_SLOTS; ++i)
228 {
229 if (!scratch_slot_init(&pool->slots[i], slot_size, i))
230 {
231 scratch_pool_destroy(pool);

233 }
234 }
235 return true;
236}

254t_arena* scratch_acquire(t_scratch_arena_pool* pool)
255{
256 if (!pool)
257 return arena_report_error(NULL, "scratch_acquire failed: pool is NULL"), NULL;
258
259 SCRATCH_LOCK(pool);
260 for (int i = 0; i < SCRATCH_MAX_SLOTS; ++i)
261 {
262 if (!atomic_exchange(&pool->slots[i].in_use, true))
263 {
264 arena_reset(&pool->slots[i].arena);
265 SCRATCH_UNLOCK(pool);
266 return &pool->slots[i].arena;
267 }
268 }
269 SCRATCH_UNLOCK(pool);
270 return arena_report_error(NULL, "scratch_acquire failed: all slots in use"), NULL;
271}

273void scratch_release(t_scratch_arena_pool* pool, t_arena* arena)
274{
275 if (!pool || !arena)
276 {
277 arena_report_error(NULL, "scratch_release failed: null pool or arena");
278 return;
279 }
280
281 for (int i = 0; i < SCRATCH_MAX_SLOTS; ++i)
282 {
283 if (&pool->slots[i].arena == arena)
284 {
285 atomic_store(&pool->slots[i].in_use, false);
286 return;
287 }
288 }
289
290 arena_report_error(arena, "scratch_release failed: arena not found in pool");
291}
