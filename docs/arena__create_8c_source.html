62t_arena* arena_create(size_t size, bool allow_grow)
63{
64 if (size == 0)
65 {
66 arena_report_error(NULL, "arena_create failed: zero size requested");
67 return NULL;
68 }
69
70 t_arena* arena = arena_alloc_struct();
71 if (!arena)

74 void* buffer = arena_alloc_buffer(size);

362static inline bool arena_init_mutex(t_arena* arena)
363{
364#ifdef ARENA_ENABLE_THREAD_SAFE
365 pthread_mutexattr_t attr;
366 if (pthread_mutexattr_init(&attr) != 0)
367 goto fail;
368 if (pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE) != 0)
369 goto fail_attr;
370 if (pthread_mutex_init(&arena->lock, &attr) != 0)
371 goto fail_attr;
372 pthread_mutexattr_destroy(&attr);
373 arena->use_lock = true;
374 return true;
375fail_attr:
376 pthread_mutexattr_destroy(&attr);
377fail:
378 arena_report_error(NULL, "arena_create failed: mutex setup failed");
379 return false;
380#else
381 (void) arena;
382 return true;
383#endif
384}

417static inline bool arena_finish_init(t_arena* arena, void* buffer, size_t size, bool allow_grow)
418{
419 arena_reset_metadata(arena);
420 arena->buffer = (uint8_t*) buffer;
421 arena->size = size;
422 atomic_store_explicit(&arena->owns_buffer, true, memory_order_release);
423 atomic_store_explicit(&arena->can_grow, allow_grow, memory_order_release);
424 atomic_store_explicit(&arena->is_destroying, false, memory_order_release);
425
426 if (!arena_init_mutex(arena))
427 {
428 free(buffer);
429 return false;
430 }
431 arena_generate_id(arena);
432 return true;
433}
